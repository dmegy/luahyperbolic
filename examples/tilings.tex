\subsection*{Gallery of triangle tilings of type $(p,q,\infty)$}

\begin{luacode*}
local PI = math.pi
local N=5
local DEPTH=8
for k=2,N do
	for l=k, N do
		if 1/k+1/l <1 then
			tex.print("\\begin{tikzpicture}[scale=1.5]")
			tex.print("\\begin{scope}")
			tex.print("\\draw[very thick] (0,0) circle (1);\\clip (0,0) circle (1);")
			local A, B, C = hyper.triangleWithAngles(PI/k, PI/l, 0)
			hyper.fillTilingFromTriangle(A, B, C, DEPTH, "even odd rule")			
			hyper.drawPolygon(A, B, C, "thick, red")
			tex.print("\\end{scope}")
			tex.print(string.format("\\draw (0,-1.2) node {$(%d, %d, \\infty)$} ;", k, l))
			tex.print("\\end{tikzpicture}")
			tex.print("~")
		end
	end
end 
\end{luacode*}


\subsection*{Gallery of triangle tilings of type $(p,q,r)$}

\begin{luacode*}
local PI = math.pi
local N=7
for k=2,N do
	for l=k, N do
	for m=l,N do
		local t = 1/k+1/l +1/m
		if t<0.99 then
			local DEPTH=math.round(15*t)
			tex.print("\\begin{tikzpicture}[scale=1.5]")
			tex.print("\\begin{scope}")
			tex.print("\\draw[very thick] (0,0) circle (1);\\clip (0,0) circle (1);")
			local A, B, C = hyper.triangleWithAngles(PI/k, PI/l, PI/m)
			hyper.fillTilingFromTriangle(A, B, C, DEPTH, "even odd rule")			
			hyper.drawPolygon(A, B, C, "thick, red")
			tex.print("\\end{scope}")
			tex.print(string.format("\\draw (0,-1.2) node {$(%d, %d, %d)$} ;", k, l,m))
			tex.print("\\end{tikzpicture}")
			tex.print("~")
		end
	end
	end
end 
\end{luacode*}
