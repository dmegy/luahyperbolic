\documentclass[margin=.2cm,multi,tikz]{standalone}
\usepackage{luahyperbolic} %loads luacode
\begin{document}
\begin{luacode*}
local A = complex(.3,.7)
local B = complex(-.8,-.3)
local C = hyper.interLL(A, B, complex.I, complex.ONE)
hyper.tikzBegin()
hyper.drawLine(complex.I, complex.ONE)
hyper.drawSegment(A,B,"very thick, red")
hyper.drawRay(C, -complex.J, "green!50!black, dashed")
hyper.drawPoints(A,B,C)
hyper.tikzEnd()
\end{luacode*}
