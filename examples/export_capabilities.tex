\documentclass[margin=.2cm,multi,tikz]{standalone}
\usepackage{luacode}
\usepackage{../luahyperbolic}
\begin{document}
\begin{luacode*}
local N=5
local DEPTH=20
for k=2,N do
	for l=k, N do
		if 1/k+1/l <1 then
			hyper.tikzBegin("scale=5")
			hyper.fillTilingFromAngles(math.pi/k,math.pi/l,0,DEPTH)
			hyper.tikzEnd(string.format("tiling-triangle-%d-%d-inf.tikz",k,l))
		end
	end
end 
\end{luacode*}
\end{document}
