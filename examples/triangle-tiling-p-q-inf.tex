\begin{document}
\directlua{
A, B, C = hyper.fundamentalIdealTriangle(\angleB,\angleC)

local geodesics = {
	{hyper.endpoints(A,B)},
	{hyper.endpoints(B,C)},
	{hyper.endpoints(C,A)}
}
geodesics = hyper.propagateGeodesics(geodesics, \depth, 0.005)
local shapes={}
local count = 0
for k,pair in ipairs(geodesics) do
	count = count + 1
	table.insert(shapes,hyper.shape_closed_line(pair[1], pair[2]))
end
shapes_string = table.concat(shapes, " ")
texio.write_nl( (os.clock() - t0) .. " s : added " .. count .. " geodesics")
}

\begin{hyperbolicdisk}[scale=5]
\fill[even odd rule]
(0,0) circle (1)
\directlua{tex.print(shapes_string)}
;
\end{hyperbolicdisk}
\begin{luacode*}

\end{luacode*}
